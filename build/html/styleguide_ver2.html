                              function updateCounter(timestamp) {
                                  if (startTime === null) {
                                      startTime = timestamp;
                                  }
          
                                  var progress = Math.min((timestamp - startTime) / duration, 1);
                                  var easedProgress = 1 - Math.pow(1 - progress, 3);
                                  var currentPercent = maxPercent * easedProgress;
          
                                  $percent.text(formatPercent(currentPercent, decimalLength));
          
                                  if (progress < 1) {
                                      requestFrame(updateCounter);
                                      return;
                                  }
          
                                  $percent.text(formatPercent(maxPercent, decimalLength));
                              }
          
                              requestFrame(updateCounter);
                          }
